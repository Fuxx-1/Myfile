class Solution {
    public String truncateSentence(String s, int k) {
		String[] str = s.split(" ");
		StringBuilder temp = new StringBuilder("");
		for (int i = 0; i < k; i++) {
			temp.append(str[i] + " ");
		}
		return temp.toString();
    }
}
